const mindee = require("mindee");
// for TS or modules:
// import * as mindee from "mindee";

const workflowId: string = "workflow-id";

// Init a new client
const mindeeClient = new mindee.v1.Client(
  { apiKey: "my-api-key" }
);

// Load a file from disk
const inputSource = new mindee.PathInput(
  { inputPath: "/path/to/the/file.ext" }
);

// Configure the RAG-enabled workflow
const workflowParams = {
  rag: true,
  workflowId: workflowId
};

// Parse the file asynchronously on a workflow queue
const asyncApiResponse = mindeeClient.enqueueAndParse(
  mindee.v1.product.FinancialDocumentV1,
  inputSource,
  workflowParams
);

// Handle the response Promise
asyncApiResponse.then((resp) => {
  // print a string summary
  console.log(resp.document?.toString());
});
